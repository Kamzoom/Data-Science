import matplotlib.pyplot as plt

# Pfad zur CSV-Datei
file_path = '/Users/kamillalauter/Datenanalyse/ds-straftaten-nach-art-seit-1990.csv'

# Laden der CSV-Datei in einen DataFrame, mit Semikolon als Trennzeichen
data = pd.read_csv(file_path, sep=';', usecols=[0, 8])

# Umbenennen der Spalten für eine einfachere Handhabung
data.columns = ['Kategorie', 'Wert']

# Erstellen des Balkendiagramms
plt.figure(figsize=(10, 6))  # Größe des Diagramms festlegen
plt.bar(data['Kategorie'], data['Wert'], color='blue')  # Balkendiagramm erstellen
plt.xlabel('Kategorie')  # X-Achsen-Beschriftung
plt.ylabel('Wert')  # Y-Achsen-Beschriftung
plt.title('Tötungsdelikte')  # Titel des Diagramms
plt.xticks(rotation=45)  # Rotation der X-Achsen-Beschriftungen für bessere Lesbarkeit
plt.tight_layout()  # Stellt sicher, dass alles gut passt
plt.show()  # Diagramm anzeigen
